# Dodaj do backend/app/main.py poniżej `from fastapi import FastAPI`:
from fastapi.middleware.cors import CORSMiddleware

# I przed wszystkimi app.include_router(...):
app.add_middleware(
    CORSMiddleware,
    allow_origins=["*"],  # lub np. ["http://localhost:5173"]
    allow_credentials=True,
    allow_methods=["*"],
    allow_headers=["*"],
)

# To pozwoli na dostęp do API z dowolnego localhost i portu (do developmentu).
